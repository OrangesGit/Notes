class Solution {
    public boolean isPerfectSquare(int num) {
        int left = 0, right = num;
        while(left <= right){
            int pivot = left + ((right - left) >> 1);
            long square = (long) pivot * pivot;
            if(square < num) left = pivot + 1;
            else if(square > num) right = pivot - 1;
            else return true;
        }
        return false;
    }
}
