487.最大连续1的个数 II

487. 最大连续1的个数 II

滑动窗口 Sliding Window

class Solution {
    public int findMaxConsecutiveOnes(int[] nums) {
        int windowStart = 0;
        int res = 0, replace_num = 1;
        for(int windowEnd = 0; windowEnd < nums.length; windowEnd++){
            int curr = nums[windowEnd];
            if(curr == 0){
                replace_num--;
                while(replace_num < 0){
                    if(nums[windowStart] == 0)
                        replace_num++;
                    windowStart++;
                }
            }
            res = Math.max(windowEnd - windowStart + 1 , res);
        }
        return res;
    }
}

$$ O(N)+O(1)

$$

DP

class Solution {
    public int findMaxConsecutiveOnes(int[] nums) {
        int n = nums.length;
        int dp0 = 0, dp1 = 0, res = 0;

        for(int i = 0; i < n; i++){
            int curr = nums[i];
            if(curr == 1){
                dp0++;
                dp1++;
            }
            else{
                dp1 = dp0+1;
                dp0 = 0;

            }
            res = Math.max(res, Math.max(dp0, dp1));
        }
        return res;
    }
}

$$ O(N)+O(1)
