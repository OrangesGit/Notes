1. 两数之和

HashMap

class Solution {
    public int[] twoSum(int[] nums, int target) {
        Map<Integer, Integer> seen = new HashMap();
        for(int i=0; i<nums.length; i++){
            int currTarget = target - nums[i];
            if(seen.containsKey(currTarget)){
                return new int[]{i,seen.get(currTarget)};
            }
            seen.put(nums[i], i);
        }
        return new int[]{};
    }
}
