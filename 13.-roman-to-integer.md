class Solution {
    public int romanToInt(String s) {
        HashMap<Character, Integer> mappings = new HashMap(){{
            put('M', 1000);
            // put("CM", 900);
            put('D', 500);
            // put("CD", 400);
            put('C', 100);
            // put("XC", 90);
            put('L', 50);
            // put("XL", 40);
            put('X', 10);
            // put("IX", 9);
            put('V', 5);
            // put("IV", 4);
            put('I', 1);
        }};
        int res = 0;
        int n = s.length();
        for(int i=n-1; i>=0; i--){
            int curr = mappings.get(s.charAt(i));
            int prev = i<n-1 ? mappings.get(s.charAt(i+1)) : Integer.MAX_VALUE;
            if(i<n-1 && curr < prev){
                res -= curr;
            }
            else{
                res += curr;
            }
        }
        return res;

    }
}
