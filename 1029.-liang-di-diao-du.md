class Solution {
    public int twoCitySchedCost(int[][] costs) {
        //将costs中 去A市的费用减去B市的费用，然后递增排序
        Arrays.sort(costs,(a,b)->{
            return Integer.compare(a[0]-a[1],b[0]-b[1]);
        });
        int n = costs.length;
        int res = 0;
        for(int i = 0; i < n; i++){
            if(i < n/2) res += costs[i][0];
            else res += costs[i][1];
        }
        return res;
    }
}
